## Pager

<Playground>
  {() => {
      const prevKey = nanoid();
      const nextKey = nanoid();

      const items = [
        {
          key: prevKey,
          element: '<'
        },
        {
          key: nanoid(),
          element: 1
        },
        {
          key: nanoid(),
          element: 2
        },
        {
          key: nanoid(),
          element: 3
        },
        {
          key: nextKey,
          element: '>'
        },
      ];

      return (
        <Kawara
          rect={{
            width: 20,
            height: 20
          }}
          gutter={5}
          items={items}
          clickable={true}
          onClick={action => {
            if (action.payload.key === prevKey) {
              console.log('prev clicked');
            }
          }}
        />
      );
    }
  }
</Playground>
